let shootPressed = false; let reloadPressed = false; let flashlightPressed = false; let weaponNext = false; let weaponPrev = false; let moveX = 0; let moveY = 0; let turnX = 0; let turnY = 0; function readController() { const pads = navigator.getGamepads(); for (let gp of pads) { if (!gp) continue; if (!gp.id.includes("DualSense")) continue; // BUTTONS shootPressed = gp.buttons[7].pressed; // R2 reloadPressed = gp.buttons[2].pressed; // Square flashlightPressed = gp.buttons[3].pressed; // Triangle weaponNext = gp.buttons[15].pressed; // D-pad right weaponPrev = gp.buttons[14].pressed; // D-pad left // STICKS moveX = gp.axes[0]; // left stick X moveY = gp.axes[1]; // left stick Y turnX = gp.axes[2]; // right stick X turnY = gp.axes[3]; // right stick Y return; } }
